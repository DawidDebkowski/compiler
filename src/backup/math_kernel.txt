#include "math_kernel.hpp"
#include "codegen.hpp"
#include <vector>
#include <string>

using namespace std;

// Hardware registers:
// 0(a): Acc, 1(b): Op1, 2(c): Op2, 3(d): RA
// 4(e): Scratch, 5(f): Scratch, 6(g): Scratch, 7(h): Addr/Temp

// Spill Memory Locations (simulate Callee-Saved registers)
// We assume memory[2..4] are safe for system use.
const long long SPILL_RE = 2;
const long long SPILL_RF = 3;
const long long SPILL_RG = 4;

void generate_mul() {
    addr_mul = code.size();
    // Input: rb (Multiplicand), rc (Multiplier)
    // Output: rb (Result)
    // Clobbers: ra, rb, rc, rd
    // Preserves: re, rf, rg (via Spill), rh
    
    // 1. Save Return Address
    emit("SWP", 3); // rd = RA
    
    // 2. Spill Scratch Registers
    // User requested "Only if there is something". The Caller must handle this.
    // For now, kernel is clobber-happy.
    // emit("RST", 0); emit("ADD", 4); emit("STORE", SPILL_RE);
    // emit("RST", 0); emit("ADD", 5); emit("STORE", SPILL_RF);
    // rg not used in MUL, but for consistency or safety we could. But MUL is hot.
    // optimize: Don't spill rg.
    
    // 3. Initialize Accumulator
    emit("RST", 4); // re = 0 (Result)
    
    long long loop_start = code.size();
    
    // 4. Check Termination (rc == 0)
    emit("RST", 0); emit("ADD", 2);
    emit("JZERO", 0); // Placeholder
    long long jump_end = code.size() - 1;
    
    // 5. Parity Check & Shift rc
    // rf = copy(rc)
    emit("RST", 0); emit("ADD", 2); emit("SWP", 5); // rf = rc
    
    // rc = rc >> 1
    emit("SHR", 2);
    
    // ra = (rc << 1)
    emit("RST", 0); emit("ADD", 2); emit("SHL", 0);
    
    // Parity = rf - ra
    emit("SWP", 5); // ra = rf, rf = (rc<<1)
    emit("SUB", 5); // ra = Old_rc - (Old_rc_even_part)
    
    emit("JZERO", 0); 
    long long jump_even = code.size() - 1;
    
    // 6. If Odd: re += rb
    emit("RST", 0); emit("ADD", 4); emit("ADD", 1); emit("SWP", 4);
    
    code[jump_even].arg = code.size();
    
    // 7. rb = rb << 1
    emit("SHL", 1);
    
    // 8. Loop
    emit("JUMP", loop_start);
    
    // 9. Cleanup
    code[jump_end].arg = code.size();
    
    // Move result re -> rb
    emit("RST", 0); emit("ADD", 4); emit("SWP", 1);
    
    // Restore Scratch Registers
    // emit("LOAD", SPILL_RE); emit("SWP", 4);
    // emit("LOAD", SPILL_RF); emit("SWP", 5);
    
    // Restore RA
    emit("RST", 0); emit("ADD", 3);
    
    emit("RTRN");
}

void generate_div() {
    addr_div = code.size();
    // Input: rb (Dividend), rc (Divisor)
    // Output: rb (Quotient), rc (Remainder)
    
    emit("SWP", 3); // Save RA
    
    // Spill All Scratch
    // Caller handles spilling.
    // emit("RST", 0); emit("ADD", 4); emit("STORE", SPILL_RE);
    // emit("RST", 0); emit("ADD", 5); emit("STORE", SPILL_RF);
    // emit("RST", 0); emit("ADD", 6); emit("STORE", SPILL_RG);
    
    // Check Zero Divisor
    emit("RST", 0); emit("ADD", 2);
    emit("JZERO", 0); 
    long long jump_zero = code.size() - 1;
    
    // Init Quotient rg = 0
    emit("RST", 6);
    // Init Multiplier rf = 1
    emit("RST", 5); emit("INC", 5);
    
    // re = rc (Divisor Copy)
    emit("RST", 0); emit("ADD", 2); emit("SWP", 4); // re=rc, ra=0
    
    // Shifts Loop
    long long loop1 = code.size();
    // if re > rb (re - rb > 0) break
    emit("RST", 0); emit("ADD", 4); emit("SUB", 1);
    emit("JPOS", 0);
    long long jump_out1 = code.size() - 1;
    
    emit("SHL", 4); // re << 1
    emit("SHL", 5); // rf << 1
    emit("JUMP", loop1);
    
    code[jump_out1].arg = code.size();
    
    // Overshot, shift back
    emit("SHR", 4);
    emit("SHR", 5);
    
    // Subtract Loop
    long long loop2 = code.size();
    // if rf == 0 break
    emit("RST", 0); emit("ADD", 5);
    emit("JZERO", 0);
    long long jump_out2 = code.size() - 1;
    
    // if re <= rb (re - rb <= 0? No. rb >= re means rb-re >= 0.
    // VM SUB is a - b. a -= b.
    // We want if re <= rb.
    // Check re > rb? re - rb > 0?
    // emit("RST", 0); emit("ADD", 4); emit("SUB", 1); -> re - rb.
    // If Result > 0, then re > rb. Skip Sub.
    
    emit("RST", 0); emit("ADD", 4); emit("SUB", 1);
    emit("JPOS", 0);
    long long jump_skip = code.size() - 1;
    
    // rb -= re
    emit("RST", 0); emit("ADD", 1); emit("SUB", 4); emit("SWP", 1); 
    // rg += rf
    emit("RST", 0); emit("ADD", 6); emit("ADD", 5); emit("SWP", 6); 
    
    code[jump_skip].arg = code.size();
    
    emit("SHR", 4); 
    emit("SHR", 5); 
    emit("JUMP", loop2);
    
    code[jump_out2].arg = code.size();
    
    // Result
    emit("RST", 0); emit("ADD", 1); emit("SWP", 2); // rc = rem (old rb)
    emit("RST", 0); emit("ADD", 6); emit("SWP", 1); // rb = quot (old rg)
    
    // emit("LOAD", SPILL_RE); emit("SWP", 4); 
    // emit("LOAD", SPILL_RF); emit("SWP", 5);
    // emit("LOAD", SPILL_RF); emit("SWP", 5);
    emit("LOAD", SPILL_RG); emit("SWP", 6);
    
    emit("RST", 0); emit("ADD", 3);
    emit("RTRN");
    
    // Zero Handler
    code[jump_zero].arg = code.size();
    emit("RST", 1); emit("RST", 2);
    // Restore even here
    // emit("LOAD", SPILL_RE); emit("SWP", 4); 
    // emit("LOAD", SPILL_RF); emit("SWP", 5);
    // emit("LOAD", SPILL_RG); emit("SWP", 6);
    emit("RST", 0); emit("ADD", 3);
    emit("RTRN");
}

void generate_mod() {
    addr_mod = code.size();
    emit("SWP", 7); // Save RA to rh (7) as div uses rd (3)
    emit("CALL", addr_div);
    emit("RST", 0);
    emit("ADD", 2); 
    emit("SWP", 1); 
    emit("RST", 0);
    emit("ADD", 7); // Restore RA from rh
    emit("RTRN");
}
